
### 片段一

```xml
  <view>
    <text>请输入第 1 个数字:</text>
    <input data-id="num1" type="number" bindchange="change3"/>
  </view>
  <view>
    <text>请输入第 2 个数字:</text>
    <input data-id="num2" type="number" bindchange="change3"/>
  </view>
  <button bindtap="compare">比较</button>
  <view>
    <block wx:if="{{num1 > num2}}">
      <text>比较结果: 第1个数大</text>
    </block>
    <block wx:elif="{{num1 < num2}}">
      <text>比较结果: 第2个数大</text>
    </block>
    <block wx:else>
      <text>比较结果: 两数相等</text>
    </block>
  </view>
```

```js
  data: {
      num1: 0,
      num2: 0,
  },
  change3: function(e) {
    var data = {}
    data[e.target.dataset.id] = Number(e.detail.value)
    this.setData(data)
  },
```

### 片段二

```xml
  <form bindsubmit="formCompare">
    <view>
      <text>请输入第 1 个数字:</text>
      <input type="number" name="num1"/>
    </view>
    <view>
      <text>请输入第 2 个数字:</text>
      <input type="number" name="num2"/>
    </view>
    <button form-type="submit">比较</button>
    <view>
      <text>比较结果:{{result}}</text>
    </view>
  </form>
```

```js
  data: {
      result: ''
  },
  formCompare: function(e) {
    var str = '两数相等'
    var n1 = Number(e.detail.value.num1)
    var n2 = Number(e.detail.value.num2)
    if (n1 > n2) {
      str = '第 1 个数大'
    } else if (n1 < n2) {
      str = '第 2 个数大'
    }
    this.setData({result: str})
  },
```
